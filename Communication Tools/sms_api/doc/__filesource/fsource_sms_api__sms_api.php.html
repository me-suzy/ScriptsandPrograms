  156.     function send($to=null, $from=null, $text=null) {

  157.  
    158. 

  158.         /* Check SMS credits balance */
    159. 

  159.         if ($this->getbalance() < $this->balace_limit) {
    160. 

  160.             die ("You have reach the SMS credit limit!");
    161. 

  161.         };
    162. 

  162.  
    163. 

  163.         /* Check SMS $text length */
    164. 

  164.         if (strlen ($text) > 465) {
    165. 

  165.             die ("Your message is to long! (Current lenght=".strlen ($text).")");
    166. 

  166.         }
    167. 

  167.  
    168. 

  168.         /* Does message need to be concatenate */

  169.         if (strlen ($text) > 160) {
    170. 

  170.             $concat = "&concat=3";

  171.         } else {
    172. 

  172.             $concat = "";
    173. 

  173.         }
    174. 

  174.  
    175. 

  175.         /* Check $to and $from is not empty */
    176. 

  176.         if (empty ($to)) {
    177. 

  177.             die ("You not specify destination address (TO)!");
    178. 

  178.         }
    179. 

  179.         if (empty ($from)) {
    180. 

  180.             die ("You not specify source address (FROM)!");
    181. 

  181.         }
    182. 

  182.  
    183. 

  183.         /* Reformat $to number */

  184.         $cleanup_chr = array ("+", " ", "(", ")", "\r", "\n", "\r\n");
    185. 

  185.         $to = str_replace($cleanup_chr, "", $to);
    186. 

  186.  
    187. 

  187.         /* Send SMS now */

  188.         $comm = sprintf ("%s/sendmsg?session_id=%s&to=%s&from=%s&text=%s&callback=%s%s",

  189.             $this->base,

  190.             $this->session,
    191. 

  191.             rawurlencode($to),
    192. 

  192.             rawurlencode($from),

  193.             rawurlencode($text),
    194. 

  194.             $this->callback,
    195. 

  195.             $concat
    196. 

  196.         );

  197.         return $this->_parse_send ($this->_execgw($comm));

  198.     }

  219.         curl_setopt ($ch, CURLOPT_HEADER, 0);

  225.         }
